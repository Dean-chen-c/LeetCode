#### 有穷性

**例子**
$$
Hailstone =\left\{
\begin{aligned}
\quad \{1\} \qquad n\leq1 \\
\quad \{n\} \cup Hailstone(n/2) \qquad n偶 \\
\quad \{n\} \cup Hailstone(3n+1) \qquad n奇 \\
\end{aligned}
\right.
$$
Hailstone(42) = { 42 , 21 , 64 , 32 , ... , 1 }

```js
/*
计算序列Hailstone(n)的长度
*/
var hailstone=function (num){
    const arr = [];
    while(1<num){
        ( num % 2 ) ? num = 3 * num + 1 : num /= 2;
        arr.push(num);
    }
    return arr;
}

hailstone(42) // [21, 64, 32, 16, 8, 4, 2, 1]
hailstone(42).length // 8

//是否有穷   现在还不知道  证明不了
```
